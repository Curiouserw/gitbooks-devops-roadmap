# 数值型的判断

# -lt    小于，如[ $a -lt $b ]
# -eq    等于，如[ $a -eq $b ]
# -ne    不等于，如[ $a -ne $b ]
# -ge    大于等于，如[ $a -ge $b ]
# le    小于等于 ，如 [ $a -le $b ]
# <     小于(需要双括号),如:(($a < $b))
# <=    小于等于(需要双括号),如:(($a <= $b))
# >     大于(需要双括号),如:(($a > $b))
# >=    大于等于(需要双括号),如:(($a >= $b))
